# 1. Initialize the local directory as a Git repository
git init

# 2. Stage all of your project files
git add .

# 3. Commit the staged files with a description
git commit -m "Initial commit"

# 4. Rename your default branch to main (standard practice)
git branch -M main

# 5. Link your local project to your remote repository URL
git remote add origin <PASTE_YOUR_REPOSITORY_URL_HERE>

# 6. Push your code and set the upstream tracking branch
git push -u origin main
